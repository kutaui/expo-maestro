```bash

```

```bash
npx expo install expo-dev-client
```

```bash
eas build --profile development-simulator --platform ios --local
```

```bash
npx expo start --dev-client
```

```bash
tar -xvzf build-*.tar.gz
```

```bash
xcrun simctl install booted maestrotutorial.app
```
